NAME
  hpq.h - library for HPQ images

SYNOPSIS
  #include "hpq.h"

  uint32_t hpq_id     = 0;
  uint32_t hpq_q      = 1;
  uint32_t hpq_x0     = 0;
  uint32_t hpq_y0     = 0;
  uint32_t hpq_width  = 256,
  uint32_t hpq_height = 256;

  _Bool *hpq_canvas;

  _Bool hpq_defined(void);

  void hpq_p0(void);
  ...
  void hpq_p39(void);

  void (*hpq_painters[40])(void);

DESCRIPTION
  HPQ images are described at https://sawuare.net/hpq-images.
